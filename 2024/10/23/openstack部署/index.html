scp /etc/yum.repos.d/local.repo compute: /etc/yum.repos.d/local.repo
并做以下修改

[centos]
name=centos
baseurl=ftp://controller/centos
gpgcheck=0
enabled=1

[iaas]
name=iaas
baseurl=ftp://controller/iaas/iaas-repo
gpgcheck=0
enabled=1
